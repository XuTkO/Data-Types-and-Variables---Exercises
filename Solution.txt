import java.util.Scanner;

public class Demo {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        String firstName = scanner.nextLine();
        String secondame = scanner.nextLine();
        int age = Integer.parseInt(scanner.nextLine());
        char gender = scanner.nextLine().charAt(0);
        long personalId = Long.parseLong(scanner.nextLine());
        int employeeNumber = Integer.parseInt(scanner.nextLine());

        System.out.printf("First name: %s%nLast name: %s%nAge: %d%nGender: %c%nPersonal ID: %d%nUnique Employee number: %d",firstName,secondame,age,gender,personalId,employeeNumber);
    }
}
